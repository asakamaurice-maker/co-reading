<script>
try {

var _ft = "";
var _books = [];
var _cur = null;
var _chs = [];
var _ci = 0;
var _an = {};
var _showA = true;
var _openP = -1;

function ls(k, v) {
  if (v === undefined) {
    try { var x = localStorage.getItem(k); return x ? JSON.parse(x) : null; } catch(e) { return null; }
  } else {
    try { localStorage.setItem(k, JSON.stringify(v)); } catch(e) {}
  }
}
function lsd(k) { try { localStorage.removeItem(k); } catch(e) {} }
function esc(s) { var d = document.createElement("div"); d.textContent = s; return d.innerHTML; }

function parse(raw) {
  var lines = raw.split(/\r?\n/);
  var out = [], t = "前言", buf = [], i;
  for (i = 0; i < lines.length; i++) {
    var tr = lines[i].trim();
    if (/^第[一二三四五六七八九十百千零0-9]+[章回节卷篇]/.test(tr)) {
      if (buf.length > 0) { var c = buf.join("\n").trim(); if (c.length > 100) out.push({t:t, c:c}); }
      t = tr; buf = [];
    } else { buf.push(lines[i]); }
  }
  if (buf.length > 0) { var c2 = buf.join("\n").trim(); if (c2.length > 100) out.push({t:t, c:c2}); }
  if (out.length === 0) out.push({t:"全文", c:raw.trim()});
  return out;
}

function fmtTime(iso) {
  try { return new Date(iso).toLocaleString("zh-CN",{month:"short",day:"numeric",hour:"2-digit",minute:"2-digit"}); }
  catch(e) { return ""; }
}

// ── Modal ──
function openModal() { document.getElementById("modal").style.display = "flex"; _ft = ""; }
function closeModal() {
  document.getElementById("modal").style.display = "none";
  document.getElementById("m-title").value = "";
  document.getElementById("m-author").value = "";
  document.getElementById("m-file").value = "";
  document.getElementById("m-info").textContent = "";
  document.getElementById("m-go").textContent = "开始阅读";
  _ft = "";
}
function onFile(el) {
  var f = el.files[0]; if (!f) return;
  var r = new FileReader();
  r.onload = function(ev) {
    _ft = ev.target.result;
    var chs = parse(_ft);
    document.getElementById("m-info").textContent = "已读取 " + _ft.length + " 字，" + chs.length + " 个章节";
  };
  r.readAsText(f, "UTF-8");
  if (!document.getElementById("m-title").value) document.getElementById("m-title").value = f.name.replace(/\.txt$/i,"");
}
function upload() {
  var title = document.getElementById("m-title").value.trim();
  var author = document.getElementById("m-author").value.trim();
  if (!title || !_ft) { alert("请填写书名并上传文件"); return; }
  var btn = document.getElementById("m-go");
  btn.textContent = "处理中..."; btn.disabled = true;
  var chs = parse(_ft);
  var id = "b" + Date.now();
  var meta = {id:id, title:title, author:author||null};
  _books.push(meta);
  ls("cr3-books", _books);
  ls("cr3-ch-" + id, chs);
  btn.textContent = "开始阅读"; btn.disabled = false;
  closeModal();
  _cur = meta; _chs = chs; _an = {}; _ci = 0; _openP = -1;
  showReader();
}

// ── Shelf ──
function showShelf() {
  _books = ls("cr3-books") || [];
  document.getElementById("shelf").style.display = "block";
  document.getElementById("reader").style.display = "none";
  var el = document.getElementById("main");
  if (_books.length === 0) {
    el.innerHTML = '<div class="empty"><div style="font-size:44px;margin-bottom:8px">📚</div><h3 style="font-family:\'Noto Serif SC\',serif;font-size:17px">书架还是空的</h3><p>上传 TXT 开始共读之旅</p><button class="pri" onclick="openModal()">上传第一本书</button></div>';
    return;
  }
  var cs = ["#8B7355","#6B7B5A","#5B6B8A","#8A5B6B","#7B6B8A"];
  var h = '<div style="font-family:\'Noto Serif SC\',serif;font-size:14px;color:#8C7B6B;letter-spacing:1px">我们的书架 · ' + _books.length + ' 本</div><div class="grid">';
  for (var i = 0; i < _books.length; i++) {
    var b = _books[i], co = cs[b.title.length % cs.length];
    h += '<div style="position:relative">';
    h += '<div class="card" style="background:linear-gradient(135deg,'+co+','+co+'BB)" onclick="openBook('+i+')">';
    h += '<div class="sp"></div><span>' + esc(b.title) + '</span>';
    if (b.author) h += '<small>' + esc(b.author) + '</small>';
    h += '</div>';
    h += '<button class="del" onclick="event.stopPropagation();delBook('+i+')">×</button></div>';
  }
  h += '</div>';
  el.innerHTML = h;
}

function openBook(idx) {
  _cur = _books[idx];
  _chs = ls("cr3-ch-" + _cur.id) || [];
  _an = ls("cr3-an-" + _cur.id) || {};
  _ci = 0; _openP = -1;
  showReader();
}

function delBook(idx) {
  if (!confirm("确定删除这本书吗？")) return;
  var b = _books[idx];
  lsd("cr3-ch-" + b.id);
  lsd("cr3-an-" + b.id);
  _books.splice(idx, 1);
  ls("cr3-books", _books);
  showShelf();
}

// ── Reader ──
function showReader() {
  document.getElementById("shelf").style.display = "none";
  var ch = _chs[_ci]; if (!ch) return;
  var paras = ch.c.split(/\n+/).filter(function(p){ return p.trim(); });
  var key = "c" + _ci;
  var ca = _an[key] || [];

  var h = '<div class="rhd">';
  h += '<button class="sm" onclick="backToShelf()">← 书架</button>';
  h += '<div class="ti"><h3>' + esc(_cur.title) + '</h3>';
  h += '<small>' + esc(ch.t) + ' · ' + (_ci+1) + '/' + _chs.length + '</small></div>';
  h += '<button class="sm' + (_showA?' on':'') + '" onclick="toggleAnno()">批注</button>';
  h += '<button class="sm" onclick="showToc()">目录</button>';
  h += '</div>';

  h += '<div id="rscroll" style="overflow-y:auto;padding:24px 14px;height:calc(100vh - 50px);height:calc(100dvh - 50px);-webkit-overflow-scrolling:touch;display:flex;justify-content:center">';
  h += '<div style="max-width:600px;width:100%">';
  h += '<h2 class="cht">' + esc(ch.t) + '</h2>';

  for (var i = 0; i < paras.length; i++) {
    var pa = ca.filter(function(a){ return a.pi === i; });
    var has = pa.length > 0;
    h += '<p class="pa' + (has?' hl':'') + '" onclick="toggleP('+i+')">' + esc(paras[i]);
    if (has) h += '<span class="bc">' + pa.length + '条</span>';
    h += '</p>';

    h += '<div class="an" id="an-'+i+'" style="display:none">';
    for (var j = 0; j < pa.length; j++) {
      var a = pa[j], ic = a.w === "claude";
      h += '<div class="bub ' + (ic?'cl':'me') + '">';
      h += '<button class="dl" onclick="event.stopPropagation();delAnno('+i+',\''+a.id+'\')">删除</button>';
      h += '<span class="nm ' + (ic?'c':'n') + '">' + (ic?'Claude':'柰柰') + '</span>';
      h += '<span class="tm">' + fmtTime(a.at) + '</span>';
      if (a.hl) h += '<div class="hl2 ' + (ic?'c':'n') + '">"' + esc(a.hl.slice(0,50)) + (a.hl.length>50?'...':'') + '"</div>';
      h += '<div class="ct">' + esc(a.tx) + '</div></div>';
    }
    h += '<div class="cr"><textarea id="ta-'+i+'" placeholder="写下你的想法..."></textarea>';
    h += '<button class="pri" style="padding:6px 10px;font-size:11px;height:32px;align-self:flex-end" onclick="addAnno('+i+')">发送</button></div>';
    h += '</div>';
  }

  h += '<div class="nv">';
  h += '<button class="btn" style="font-size:12px;opacity:' + (_ci<=0?'.3':'1') + '" onclick="goCh('+(_ci-1)+')">← 上一章</button>';
  h += '<span style="font-size:11px;color:#8C7B6B">' + (_ci+1) + '/' + _chs.length + '</span>';
  h += '<button class="btn" style="font-size:12px;opacity:' + (_ci>=_chs.length-1?'.3':'1') + '" onclick="goCh('+(_ci+1)+')">下一章 →</button>';
  h += '</div></div></div>';

  document.getElementById("reader").innerHTML = h;
  document.getElementById("reader").style.display = "block";
}

function backToShelf() {
  document.getElementById("reader").style.display = "none";
  showShelf();
}

function toggleP(i) {
  if (!_showA) return;
  if (_openP >= 0) {
    var old = document.getElementById("an-" + _openP);
    if (old) old.style.display = "none";
  }
  if (_openP === i) { _openP = -1; return; }
  _openP = i;
  var el = document.getElementById("an-" + i);
  if (el) el.style.display = "block";
}

function toggleAnno() {
  _showA = !_showA;
  _openP = -1;
  showReader();
}

function addAnno(pi) {
  var ta = document.getElementById("ta-" + pi);
  if (!ta || !ta.value.trim()) return;
  var ch = _chs[_ci];
  var paras = ch.c.split(/\n+/).filter(function(p){ return p.trim(); });
  var key = "c" + _ci;
  if (!_an[key]) _an[key] = [];
  _an[key].push({
    id: "a" + Date.now(),
    w: "nainai",
    tx: ta.value.trim(),
    hl: (paras[pi]||"").slice(0,80),
    pi: pi,
    at: new Date().toISOString()
  });
  ls("cr3-an-" + _cur.id, _an);
  var savedP = _openP;
  showReader();
  _openP = savedP;
  var el = document.getElementById("an-" + savedP);
  if (el) el.style.display = "block";
}

function delAnno(pi, aid) {
  var key = "c" + _ci;
  if (!_an[key]) return;
  _an[key] = _an[key].filter(function(a){ return a.id !== aid; });
  ls("cr3-an-" + _cur.id, _an);
  var savedP = _openP;
  showReader();
  _openP = savedP;
  var el = document.getElementById("an-" + savedP);
  if (el) el.style.display = "block";
}

function goCh(n) {
  if (n < 0 || n >= _chs.length) return;
  _ci = n; _openP = -1;
  showReader();
  var sc = document.getElementById("rscroll");
  if (sc) sc.scrollTop = 0;
}

// ── TOC ──
function showToc() {
  var h = '<div style="display:flex;justify-content:space-between;margin-bottom:10px"><h3>目录</h3><button onclick="hideToc()" style="background:none;border:none;cursor:pointer;color:#8C7B6B;font-size:16px">×</button></div>';
  for (var i = 0; i < _chs.length; i++) {
    h += '<div class="toc-i' + (i===_ci?' cur':'') + '" onclick="goCh('+i+');hideToc()">' + esc(_chs[i].t) + '</div>';
  }
  document.getElementById("toc").innerHTML = h;
  document.getElementById("toc").style.display = "block";
  document.getElementById("toc-bg").style.display = "block";
}
function hideToc() {
  document.getElementById("toc").style.display = "none";
  document.getElementById("toc-bg").style.display = "none";
}

// Make all functions global
window.openModal=openModal; window.closeModal=closeModal; window.onFile=onFile; window.upload=upload;
window.showShelf=showShelf; window.openBook=openBook; window.delBook=delBook;
window.backToShelf=backToShelf; window.toggleP=toggleP; window.toggleAnno=toggleAnno;
window.addAnno=addAnno; window.delAnno=delAnno; window.goCh=goCh;
window.showToc=showToc; window.hideToc=hideToc;

showShelf();

} catch(e) { document.body.innerHTML = '<pre style="color:red;padding:20px">错误: ' + e.message + '\n' + e.stack + '</pre>'; }
</script>
